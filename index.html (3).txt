    <script>
        let posts = [];
        let isLoggedIn = false;

        function hideIntro() {
            const introScreen = document.getElementById('introScreen');
            const mainContent = document.getElementById('mainContent');
            
            introScreen.classList.add('hide');
            
            setTimeout(() => {
                introScreen.style.display = 'none';
                mainContent.style.display = 'block';
            }, 1000);
        }

        function showLoginForm() {
            document.getElementById('loginPanel').style.display = 'none';
            document.getElementById('loginForm').style.display = 'block';
        }

        function login() {
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;

            if (username === 'usuario' && password === '098765') {
                isLoggedIn = true;
                document.getElementById('loginForm').style.display = 'none';
                document.getElementById('adminPanel').style.display = 'block';
                loadPosts();
            } else {
                alert('Credenciales incorrectas');
            }
        }

        function toggleContentFields() {
            const contentType = document.getElementById('contentType').value;
            const textContent = document.getElementById('textContent');
            const imageContent = document.getElementById('imageContent');
            const linkContent = document.getElementById('linkContent');

            // Ocultar todos los campos
            textContent.style.display = 'none';
            imageContent.style.display = 'none';
            linkContent.style.display = 'none';

            // Mostrar campos según el tipo
            switch(contentType) {
                case 'text':
                    textContent.style.display = 'block';
                    break;
                case 'image':
                    imageContent.style.display = 'block';
                    break;
                case 'link':
                    linkContent.style.display = 'block';
                    break;
                case 'mixed':
                    textContent.style.display = 'block';
                    imageContent.style.display = 'block';
                    linkContent.style.display = 'block';
                    break;
            }
        }

        function createPost() {
            if (!isLoggedIn) return;

            const title = document.getElementById('postTitle').value;
            const content = document.getElementById('postContent').value;
            const image = document.getElementById('postImage').value;
            const link = document.getElementById('postLink').value;
            const contentType = document.getElementById('contentType').value;

            if (!title) {
                alert('Por favor ingresa un título');
                return;
            }

            const post = {
                id: Date.now(),
                title: title,
                content: content,
                image: image,
                link: link,
                type: contentType,
                date: new Date().toLocaleString('es-ES')
            };

            posts.unshift(post);
            
            // Limpiar formulario
            document.getElementById('postTitle').value = '';
            document.getElementById('postContent').value = '';
            document.getElementById('postImage').value = '';
            document.getElementById('postLink').value = '';

            renderPosts();
        }

        function deletePost(id) {
            if (!isLoggedIn) return;
            
            posts = posts.filter(post => post.id !== id);
            renderPosts();
        }

        function renderPosts() {
            const container = document.getElementById('postsContainer');
            
            if (posts.length === 0) {
                container.innerHTML = '<div style="text-align: center; color: #999; padding: 50px;">No hay publicaciones aún</div>';
                return;
            }

            container.innerHTML = posts.map(post => `
                <div class="post">
                    <div class="post-title">${post.title}</div>
                    
                    ${post.content ? `<div class="post-content">${post.content}</div>` : ''}
                    
                    ${post.image ? `<img src="${post.image}" alt="Imagen del post" class="post-image" onerror="this.style.display='none'">` : ''}
                    
                    ${post.link ? `<div class="post-content"><a href="${post.link}" target="_blank" class="post-link">🔗 ${post.link}</a></div>` : ''}
                    
                    <div class="post-date">
                        ${post.date}
                        ${isLoggedIn ? `<button class="delete-btn" onclick="deletePost(${post.id})">Eliminar</button>` : ''}
                    </div>
                </div>
            `).join('');
        }

        function loadPosts() {
            // Algunos posts de ejemplo
            posts = [

                    content: "Este es un espacio donde compartimos contenido aleatorio, interesante y divertido. ¡Explora y descubre cosas nuevas!",

                },

                    type: "text",
                    date: new Date().toLocaleString('es-ES')
                }
            ];
            
            renderPosts();
        }
